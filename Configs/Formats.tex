  (aux1) -- coordinate (a)
  ++(225:5) --
  ++(-45:5.1) coordinate (b);
\draw[shorten <= -10pt]
  (aux3) --
  (a) --
  (aux1);
\draw[opacity=0.6,titlepagecolor,shorten <= -10pt]
  (b) --
  ++(225:2.2) --
  ++(-45:2.2);

  (aux4) --
  ++(225:0.8) --
  ++(-45:0.8);

  (aux2) --

  (aux2) --

 {\endMakeFramed}

  [0.3em]{\Large\bfseries\vspace*{2\baselineskip}}
  {\hspace{-1.2cm}\parbox{6.5em}{%

  {}{\endleftbar}
